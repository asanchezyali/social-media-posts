% highlight: accent colour + a hand-drawn double underline (two quick pen
% strokes: slight curve, slight slope, not parallel -> organic, not a ruler)
\newcommand{\hl}[1]{%
  \tikz[baseline=(u.base)]{%
    \node[inner sep=0pt, outer sep=0pt] (u) {\textcolor{accent}{#1}};%
    \coordinate (a) at ([yshift=-1.5pt]u.south west);
    \coordinate (b) at ([yshift=-2.2pt]u.south east);
    \coordinate (c) at ([yshift=-3.4pt]u.south west);
    \coordinate (d) at ([yshift=-2.9pt]u.south east);
    \draw[accent, line width=0.7pt, line cap=round]
      (a) .. controls ($(a)!.35!(b)+(0,-0.5pt)$) and ($(a)!.7!(b)+(0,0.3pt)$) .. (b);
    \draw[accent, line width=0.55pt, line cap=round]
      (c) .. controls ($(c)!.3!(d)+(0,0.5pt)$) and ($(c)!.75!(d)+(0,-0.3pt)$) .. (d);
  }%
}

% ---- lecture-notes structure ----------------------------------------
% topic heading: handwritten, underlined with two quick pen strokes —
% the way a student actually heads a section in their notebook.
% section heading: handwritten title with a highlighter-marker swipe behind
% it (the way a student marks a section header).
\newcounter{topicnum}
% The article FLOWS and fills each page like a normal document (no one-topic-
% per-slide gaps).  A tiny needspace only stops a heading from landing in the
% last few mm of a page; the \nobreak glue after the heading keeps it welded to
% its first following line, so a heading is never left orphaned at a page foot.
%
% Optional arg = needspace reserved before the heading (default 4mm).  For a
% topic that ends in a LARGE atomic figure (pgfplots graph, \eqfit matrix), set
% it to the whole topic's height, e.g. \topic[84mm]{...}, so the entire block
% (heading + text + figure) moves to the next page together instead of the
% figure tearing off onto its own slide.
\newcommand{\topic}[2][4mm]{\par\addvspace{3mm}\needspace{#1}\refstepcounter{topicnum}%
  \noindent\begin{tikzpicture}
    \node[anchor=base west, inner sep=0pt] (tt) at (0,0)
      {\handfont\fontsize{19}{23}\selectfont\color{ink}#2};
    \begin{scope}[on background layer]
      \fill[hltopic, opacity=0.45, rounded corners=0.8mm]
        ([xshift=-1mm,yshift=-1mm]tt.south west)
        rectangle ([xshift=1mm,yshift=0.6mm]tt.north east);
    \end{scope}
  \end{tikzpicture}\par\nobreak\vskip 2.5mm\nobreak}
% single hand-drawn underline (one pen stroke, slight curve — not a ruler)
\newcommand{\uhand}[1]{%
  \tikz[baseline=(u.base)]{%
    \node[inner sep=0pt, outer sep=0pt] (u) {{\color{ink}#1}};%
    \coordinate (a) at ([yshift=-1.6pt]u.south west);
    \coordinate (b) at ([yshift=-2.3pt]u.south east);
    \draw[ink, line width=0.6pt, line cap=round]
      (a) .. controls ($(a)!.35!(b)+(0,-0.5pt)$) and ($(a)!.7!(b)+(0,0.3pt)$) .. (b);
  }%
}

% display equations -- all at ONE uniform size (break wide ones by hand).
% Centred with clear breathing room ABOVE and BELOW so they never crowd the text.
% Tune \eqskip to change that breathing room everywhere.
\newlength{\eqskip}\setlength{\eqskip}{4mm}
\newcommand{\eq}[1]{\par\addvspace{\eqskip}\centerline{$\displaystyle\color{eqink}#1$}\addvspace{\eqskip}}
\newcommand{\eqx}[1]{\par\addvspace{\eqskip}\centerline{$\displaystyle\color{eqink}#1$}\addvspace{\eqskip}}
% for big blocks (matrices) that may be scaled to fit the width.
% \centerline centres it even under \RaggedRight (adjustbox alone would sit left).
\newcommand{\eqfit}[1]{\par\addvspace{\eqskip}%
  \centerline{\adjustbox{max width=0.98\linewidth}{$\displaystyle\color{eqink}#1$}}\addvspace{\eqskip}}
% key result boxed by hand — four slightly-wobbly pen strokes around the
% equation, like circling the result you'll need for the exam.
\newcommand{\keyeq}[1]{\par\addvspace{\eqskip}%
  \begin{center}\begin{tikzpicture}
    \node[inner xsep=3.4mm, inner ysep=2.6mm] (k) {$\displaystyle\color{eqink}#1$};
    \coordinate (nw) at ([shift={( 0.5pt,-0.4pt)}]k.north west);
    \coordinate (ne) at ([shift={(-0.6pt,-1.1pt)}]k.north east);
    \coordinate (se) at ([shift={(-0.3pt, 0.9pt)}]k.south east);
    \coordinate (sw) at ([shift={( 0.7pt, 0.5pt)}]k.south west);
    \draw[accent, line width=0.75pt, line cap=round]
      (nw) .. controls ($(nw)!.4!(ne)+(0,0.7pt)$) and ($(nw)!.75!(ne)+(0,-0.4pt)$) .. (ne)
      (ne) .. controls ($(ne)!.5!(se)+(0.6pt,0)$) .. (se)
      (se) .. controls ($(se)!.45!(sw)+(0,-0.6pt)$) and ($(se)!.8!(sw)+(0,0.4pt)$) .. (sw)
      (sw) .. controls ($(sw)!.5!(nw)+(-0.6pt,0)$) .. (nw);
  \end{tikzpicture}\end{center}\addvspace{\eqskip}}

% ---- full-bleed cover photo with a uniform veil --------------------
%   Both the photo and the cream veil are placed on THIS page's background via
%   eso-pic (\AtPageLowerLeft), so the cover is reliably full-bleed even on a
%   SINGLE XeLaTeX pass.  Two deliberate choices make it robust:
%     * NOT a remember-picture overlay (that only resolves on the 2nd pass and
%       otherwise collapses the photo to a top band).
%     * The veil is a pre-baked alpha PNG (../Headers/veil.png, cream @ ~0.66),
%       NOT a pgf `opacity` fill — pgf transparency corrupts the eso-pic
%       background stream here (unknown ExtGState) and blanks the whole cover.
%   The optional arg is accepted for backwards compatibility but the veil tint
%   is fixed by veil.png (regenerate it to change the tint/opacity).
\newcommand{\coverphoto}[2][0.66]{%
  \AddToShipoutPictureBG*{%
    \AtPageLowerLeft{\includegraphics[width=\paperwidth,height=\paperheight,%
      keepaspectratio=false]{#2}}%
    \AtPageLowerLeft{\includegraphics[width=\paperwidth,height=\paperheight,%
      keepaspectratio=false]{\yalixroot Headers/veil.png}}%
  }}
