% -------------------------------------------------- motivación -----
\topic{De sumar y escalar a una estructura}
\hand{Un vector es, informalmente, algo que se puede \hl{sumar} y \hl{escalar} sin salirse de su tipo. Para formalizarlo necesitamos primero la estructura mínima: el grupo.}

% -------------------------------------------------- grupo ----------
\topic{Qué es un grupo}
\hand{Un \uhand{grupo} $(\mathcal{G},\otimes)$ es un conjunto con una operación que respeta cuatro reglas:}
\begin{center}
\begin{tikzpicture}
  \node[draw=cellEdge, line width=0.4pt, fill=cellBase, rounded corners=1.4mm,
        text width=76mm, inner sep=2.4mm, font=\small\color{ink}] {%
    \textbf{1.\ Cerradura:}\ \ $x\otimes y\in\mathcal{G}$\\[1.4mm]
    \textbf{2.\ Asociatividad:}\ \ $(x\otimes y)\otimes z = x\otimes(y\otimes z)$\\[1.4mm]
    \textbf{3.\ Neutro:}\ \ $\exists\,e:\ x\otimes e = e\otimes x = x$\\[1.4mm]
    \textbf{4.\ Inverso:}\ \ $\exists\,x^{-1}:\ x\otimes x^{-1}=e$};
\end{tikzpicture}
\end{center}
\obs{Si además $x\otimes y=y\otimes x$, el grupo es \hl{abeliano} (conmutativo).}

% -------------------------------------------------- advertencia ----
\topic{Una advertencia}
\obs{El inverso $x^{-1}$ lo es respecto a $\otimes$: \hl{no} significa $\tfrac{1}{x}$. En $(\mathbb{Z},+)$, por ejemplo, el inverso de $x$ es $-x$.}

% -------------------------------------------------- ejemplos -------
\topic{¿Cuáles son grupos?}
\begin{center}

  \node[anchor=west, font=\small\color{ink}] at (0,-3) {$(\mathbb{R}\setminus\{0\},\cdot)$};

\obs{Las matrices invertibles $n\times n$ bajo $\cdot$ forman el \hl{grupo lineal general} $GL(n,\mathbb{R})$; no es abeliano porque el producto de matrices no conmuta.}

% -------------------------------------------------- espacio vect. --
\topic{El espacio vectorial}
\hand{Súmale escalares a un grupo abeliano y obtienes un espacio vectorial:}
\defn{Un \uhand{espacio vectorial} $(V,+,\cdot)$ es un grupo abeliano $(V,+)$ junto con un producto externo $\cdot:\mathbb{R}\times V\to V$.}
\hand{El escalar debe cumplir distributividad, asociatividad y neutro:}
\eqx{\lambda(\mathbf{x}+\mathbf{y})=\lambda\mathbf{x}+\lambda\mathbf{y}
     \qquad(\lambda+\psi)\mathbf{x}=\lambda\mathbf{x}+\psi\mathbf{x}}
\eqx{\lambda(\psi\mathbf{x})=(\lambda\psi)\mathbf{x}\qquad 1\cdot\mathbf{x}=\mathbf{x}}
\obs{Los elementos $\mathbf{x}\in V$ son los \hl{vectores}; el neutro de $(V,+)$ es el vector cero $\mathbf{0}$.}

% -------------------------------------------------- ejemplos esp. --
\topic{Ejemplos de espacios}
\hand{Con la suma y el escalado usuales son espacios vectoriales $\mathbb{R}^n$, las matrices $\mathbb{R}^{m\times n}$ y los complejos $\mathbb{C}$.}
\obs{Ojo: la operación \hl{interna} ($+$) y la \hl{externa} ($\cdot$) no son los productos interno/externo. Y $\mathbf{a}\mathbf{b}$ no existe: solo el externo $\mathbf{a}\mathbf{b}^\top$ y el interno $\mathbf{a}^\top\mathbf{b}$.}

% -------------------------------------------------- columna/fila ---
\topic{Vector columna y fila}
\defn{Un \uhand{vector columna} es lo que $\mathbf{x}$ denota por defecto; su transpuesta $\mathbf{x}^\top$ es un vector fila.}
\eqfit{\mathbf{x}=\begin{bmatrix}x_1\\\vdots\\x_n\end{bmatrix}\in\mathbb{R}^{n\times1}
       \qquad
       \mathbf{x}^\top=[\,x_1\ \cdots\ x_n\,]\in\mathbb{R}^{1\times n}}
\obs{$\mathbb{R}^n$, $\mathbb{R}^{n\times1}$ y $\mathbb{R}^{1\times n}$ son el mismo espacio; solo cambia la escritura.}
