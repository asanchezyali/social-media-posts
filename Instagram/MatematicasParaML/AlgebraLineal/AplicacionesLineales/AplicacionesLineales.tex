% -------------------------------------------------- linealidad -----
\topic{Respetar la estructura}
\hand{Una \uhand{aplicación lineal} $\Phi:V\to W$ es la que respeta la suma y el escalado: sumar y luego mapear da lo mismo que mapear y luego sumar.}
\keyeq{\Phi(\lambda\mathbf{x}+\psi\mathbf{y}) = \lambda\,\Phi(\mathbf{x}) + \psi\,\Phi(\mathbf{y})}
\begin{center}
\begin{tikzpicture}[x=8.5mm,y=8.5mm]
  % ---- cuadrícula original (cuadrada) ----
  \draw[cellEdge!70, line width=0.3pt] (0,0) grid (2,2);
  \draw[-{Latex[length=1.6mm]}, eqink, line width=0.9pt] (0,0)--(1,0);
  \draw[-{Latex[length=1.6mm]}, eqink, line width=0.9pt] (0,0)--(0,1);
  \fill[ink] (0,0) circle (0.7pt);
  \node[font=\scriptsize\color{eqink}] at (0.6,-0.45){$\mathbf{e}_1$};
  \node[font=\scriptsize\color{eqink}] at (-0.5,0.6){$\mathbf{e}_2$};
  % ---- flecha Phi ----
  \draw[-{Latex[length=1.8mm]}, ink, line width=0.8pt] (2.5,1)--(3.7,1);
  \node[font=\small\color{ink}] at (3.1,1.5){$\Phi$};
  % ---- cuadrícula transformada (paralelogramos): e1'=(1,0.35) e2'=(0.5,1) ----
  \begin{scope}[shift={(4.4,0)}]
    \foreach \i in {0,1,2}{\draw[cellEdge!70, line width=0.3pt]
      (\i,{0.35*\i}) -- ({\i+1},{0.35*\i+2});}
    \foreach \j in {0,1,2}{\draw[cellEdge!70, line width=0.3pt]
      ({0.5*\j},\j) -- ({2+0.5*\j},{0.7+\j});}
    \draw[-{Latex[length=1.6mm]}, gold, line width=0.9pt] (0,0)--(1,0.35);
    \draw[-{Latex[length=1.6mm]}, gold, line width=0.9pt] (0,0)--(0.5,1);
    \fill[ink] (0,0) circle (0.7pt);
    \node[font=\scriptsize\color{gold}] at (1.15,0.1){$\Phi\mathbf{e}_1$};
    \node[font=\scriptsize\color{gold}] at (0.1,1.15){$\Phi\mathbf{e}_2$};
  \end{scope}
\end{tikzpicture}
\end{center}
\hand{Deforma la cuadrícula sin \hl{romperse}: las rectas siguen rectas y el origen no se mueve.}

% -------------------------------------------------- inversa/nombres
\topic{Deshacer y dar nombres}
\hand{Una $\Phi$ biyectiva se puede \hl{deshacer}: existe su inversa $\Phi^{-1}$ con $\Phi^{-1}\!\circ\Phi=\mathrm{id}$. Y según a dónde va y si es biyectiva, cambia de nombre:}
\begin{center}
\begin{tikzpicture}
  \node[draw=cellEdge, line width=0.4pt, fill=cellBase, rounded corners=1.4mm,
        text width=74mm, inner sep=2.6mm, font=\small\color{ink}] {%
    \textbf{Isomorfismo:}\ lineal y biyectiva, $V\to W$.\\[1.5mm]
    \textbf{Endomorfismo:}\ lineal, $V\to V$.\\[1.5mm]
    \textbf{Automorfismo:}\ lineal y biyectiva, $V\to V$.\\[1.5mm]
    \textbf{Identidad:}\ $\mathrm{id}_V(\mathbf{x})=\mathbf{x}$, el automorfismo que no hace nada.};
\end{tikzpicture}
\end{center}

% -------------------------------------------------- teorema --------
\topic{Misma dimensión, misma cosa}
\hand{Dos espacios de dimensión finita son \hl{isomorfos} —hay una biyección lineal entre ellos— exactamente cuando tienen la misma dimensión:}
\keyeq{V\cong W \;\iff\; \dim(V)=\dim(W)}
\obs{Por eso $\mathbb{R}^2$ y $\mathbb{C}$ son "lo mismo" ($z=x_1+ix_2$), y las matrices $\mathbb{R}^{m\times n}$ equivalen a $\mathbb{R}^{mn}$: misma dimensión, transformables sin \hl{pérdida}.}
