% -------------------------------------------------- definiciones ---
\topic{Dos subespacios que trae $\Phi$}
\hand{Toda aplicación lineal $\Phi:V\to W$ carga consigo dos subespacios:}
\defn{El \uhand{núcleo} (o espacio nulo) son los vectores que $\Phi$ envía al cero; la \uhand{imagen} (o rango), los que $\Phi$ alcanza.}
\eqx{\ker(\Phi)=\{\mathbf{v}\in V:\ \Phi(\mathbf{v})=\mathbf{0}\}}
\eqx{\operatorname{Im}(\Phi)=\{\Phi(\mathbf{v}):\ \mathbf{v}\in V\}}
\obs{Llamamos \hl{dominio} a $V$ y \hl{codominio} a $W$.}

\topic{El mapa completo}

\obs{Como $\Phi(\mathbf{0}_V)=\mathbf{0}_W$, el $\mathbf{0}$ siempre está en el núcleo. Todo el $\ker\Phi$ colapsa al $\mathbf{0}_W$; la $\operatorname{Im}\Phi$ es lo que de verdad se alcanza.}
\obs{Ambos son subespacios, y $\Phi$ es inyectiva exactamente cuando $\ker\Phi=\{\mathbf{0}\}$.}

% -------------------------------------------------- matrices ------
\topic{Con una matriz}
\hand{Si $\Phi(\mathbf{x})=A\mathbf{x}$ y $A$ tiene columnas $\mathbf{a}_1,\dots,\mathbf{a}_n$, la imagen es su espacio \hl{columna}:}
\eqx{\operatorname{Im}(\Phi)=\operatorname{span}[\mathbf{a}_1,\dots,\mathbf{a}_n]}
\obs{Como $\operatorname{rk}(A)=\dim(\operatorname{Im}\Phi)$, la imagen mide justo el rango de $A$.}
\obs{Y el núcleo es el conjunto solución de $A\mathbf{x}=\mathbf{0}$: codifica cómo dependen entre sí las columnas.}

% -------------------------------------------------- ejemplo -------
\topic{Un ejemplo}
\hand{Sea $\Phi:\mathbb{R}^4\to\mathbb{R}^2$, $\ \mathbf{x}\mapsto A\mathbf{x}$ con $A=\big(\begin{smallmatrix}1&2&-1&0\\1&0&0&1\end{smallmatrix}\big)$. La imagen es el span de sus columnas.}
\hand{Para el núcleo, llevamos $A$ a la forma reducida:}
\begin{center}
\begin{tikzpicture}[x=6.4mm,y=6.4mm]
  \cell{0}{0}{hlA}{$1$}\cell{1}{0}{cellBase}{$0$}\cell{2}{0}{cellBase}{$0$}\cell{3}{0}{cellBase}{$1$}
  \cell{0}{-1}{cellBase}{$0$}\cell{1}{-1}{hlA}{$1$}\cell{2}{-1}{cellBase}{$-\tfrac12$}\cell{3}{-1}{cellBase}{$-\tfrac12$}
\end{tikzpicture}
\end{center}
\hand{Sus dos columnas libres dan una base del núcleo (aquí $\dim\ker=2$):}
\eqfit{\ker(\Phi)=\operatorname{span}\!\left[\begin{bmatrix}0\\ \tfrac12\\ 1\\ 0\end{bmatrix},\ \begin{bmatrix}-1\\ \tfrac12\\ 0\\ 1\end{bmatrix}\right]}

% -------------------------------------------------- rango-nulidad -
\topic{Rango y nulidad}
\hand{El dominio se reparte \hl{exactamente} entre lo que se aplasta y lo que se alcanza —el teorema fundamental de las aplicaciones lineales:}
\keyeq{\dim(\ker\Phi)+\dim(\operatorname{Im}\Phi)=\dim(V)}
\obs{Consecuencia elegante: si $\dim V=\dim W$, ser inyectiva, sobreyectiva o biyectiva es \hl{lo mismo}.}
