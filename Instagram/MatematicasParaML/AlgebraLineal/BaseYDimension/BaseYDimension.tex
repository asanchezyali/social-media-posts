% -------------------------------------------------- span -----------
\topic{Generar y el span}
\hand{Un conjunto $\mathcal{A}=\{\mathbf{x}_1,\dots,\mathbf{x}_k\}$ \uhand{genera} $V$ si \emph{todo} vector de $V$ es una combinación lineal de sus elementos.}
\obs{El \hl{span} (o espacio generado) es el conjunto de \emph{todas} esas combinaciones. Si $\mathcal{A}$ genera $V$, escribimos:}
\keyeq{V = \operatorname{span}[\mathcal{A}]}

% -------------------------------------------------- base -----------
\topic{Qué es una base}
\hand{Una \uhand{base} es el generador \hl{mínimo}: si le quitas un vector, ya no genera $V$.}
\obs{Equivale a un generador que además es linealmente \hl{independiente}. Ese equilibrio —justo lo suficiente, sin sobrar— es lo que la hace especial.}

% -------------------------------------------------- tres caras -----
\topic{Tres caras de una base}
\hand{$B$ es una base de $V$ si cumple cualquiera de estas (todas equivalentes):}
\begin{center}
\begin{tikzpicture}
  \node[draw=cellEdge, line width=0.4pt, fill=cellBase, rounded corners=1.4mm,
        text width=74mm, inner sep=2.6mm, font=\small\color{ink}] {%
    \textbf{1.\ Generador mínimo:}\ no le sobra ninguno.\\[1.5mm]
    \textbf{2.\ Independiente máximo:}\ no cabe uno más sin crear dependencia.\\[1.5mm]
    \textbf{3.\ Representación única:}\ cada vector se escribe de una sola forma como combinación de $B$.};
\end{tikzpicture}
\end{center}

% -------------------------------------------------- dimensión ------
\topic{La dimensión}
\hand{Todas las bases de $V$ tienen el \hl{mismo} tamaño. Ese número es la \uhand{dimensión}:}
\keyeq{\dim(V) = \#\,\text{vectores de una base}}
\obs{Es el número de direcciones \hl{independientes}, no de componentes: una recta en $\mathbb{R}^2$ tiene dimensión $1$ aunque sus vectores tengan dos entradas.}
\obs{Y para un subespacio $U\subseteq V$: $\dim(U)\le\dim(V)$, con igualdad solo si $U=V$.}

% -------------------------------------------------- hallar base ----
\topic{Cómo hallar una base}
\hand{Pon los vectores como \hl{columnas}, lleva la matriz a la forma escalonada y quédate con las columnas \hl{pivote}:}
\begin{center}
\begin{tikzpicture}[x=6.6mm, y=6.6mm]
  \cell{0}{0}{hlA}{$1$}\cell{1}{0}{hlA}{$2$}\cell{2}{0}{hlB}{$3$}\cell{3}{0}{hlA}{$-1$}
  \cell{0}{-1}{hlA}{$0$}\cell{1}{-1}{hlA}{$1$}\cell{2}{-1}{hlB}{$2$}\cell{3}{-1}{hlA}{$-2$}
  \cell{0}{-2}{hlA}{$0$}\cell{1}{-2}{hlA}{$0$}\cell{2}{-2}{hlB}{$0$}\cell{3}{-2}{hlA}{$1$}
  \cell{0}{-3}{hlA}{$0$}\cell{1}{-3}{hlA}{$0$}\cell{2}{-3}{hlB}{$0$}\cell{3}{-3}{hlA}{$0$}
  \node[font=\scriptsize\color{ink}] at (0,0.95){$\mathbf{x}_1$};
  \node[font=\scriptsize\color{ink}] at (1,0.95){$\mathbf{x}_2$};
  \node[font=\scriptsize\color{ink}] at (2,0.95){$\mathbf{x}_3$};
  \node[font=\scriptsize\color{ink}] at (3,0.95){$\mathbf{x}_4$};
\end{tikzpicture}
\end{center}
\obs{Los pivotes caen en $\mathbf{x}_1,\mathbf{x}_2,\mathbf{x}_4$ (aqua): esos tres son una base del subespacio. La columna sin pivote (oro) es combinación de las anteriores.}
