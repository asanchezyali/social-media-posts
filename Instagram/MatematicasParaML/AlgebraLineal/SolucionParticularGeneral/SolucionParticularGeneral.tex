% -------------------------------------------------- el problema -----
\topic{El problema}
\hand{Un sistema $A\mathbf{x}=\mathbf{b}$ con más \hl{incógnitas} que ecuaciones suele tener \hl{infinitas} soluciones. ¿Cómo las describimos todas de golpe?}
\eqfit{\begin{bmatrix}1&0&8&-4\\0&1&2&12\end{bmatrix}
   \begin{bmatrix}x_1\\x_2\\x_3\\x_4\end{bmatrix}
   = \begin{bmatrix}42\\8\end{bmatrix}}
\obs{$2$ ecuaciones, $4$ incógnitas. Las dos primeras columnas son $\mathbf{e}_1$ y $\mathbf{e}_2$: eso nos regala el trabajo.}

% -------------------------------------------------- particular ------
\topic{Una solución particular}
\hand{Como las dos primeras columnas son $\mathbf{e}_1$ y $\mathbf{e}_2$, el lado derecho se arma directo:}
\eq{\mathbf{b}=\begin{bmatrix}42\\8\end{bmatrix}
   = 42\begin{bmatrix}1\\0\end{bmatrix}+8\begin{bmatrix}0\\1\end{bmatrix}}
\keyeq{\mathbf{x}_p = (42,\,8,\,0,\,0)}
\obs{Usa solo las columnas \hl{pivote} y pone en cero las demás.}

% -------------------------------------------------- idea clave ------
\topic{La idea clave}
\hand{Si $\mathbf{x}_p$ resuelve $A\mathbf{x}=\mathbf{b}$ y $\mathbf{z}$ resuelve $A\mathbf{x}=\mathbf{0}$, entonces su suma también resuelve $A\mathbf{x}=\mathbf{b}$:}
\eq{A(\mathbf{x}_p+\mathbf{z}) = \underbrace{A\mathbf{x}_p}_{\mathbf{b}} + \underbrace{A\mathbf{z}}_{\mathbf{0}} = \mathbf{b}}
\hand{Por eso, la clave de todo:}
\keyeq{\text{general} = \text{particular} + \text{núcleo de }A}

% -------------------------------------------------- núcleo ----------
\topic{Generar el cero}
\hand{Para las soluciones de $A\mathbf{x}=\mathbf{0}$, escribimos cada columna \hl{libre} con las pivote. La columna $3$:}
\eq{\mathbf{c}_3=\begin{bmatrix}8\\2\end{bmatrix}=8\,\mathbf{c}_1+2\,\mathbf{c}_2}
\eq{\Rightarrow\quad 8\,\mathbf{c}_1+2\,\mathbf{c}_2-\mathbf{c}_3=\mathbf{0}}
\hand{Esa relación es una solución de $A\mathbf{x}=\mathbf{0}$. Lo mismo con la columna $4$:}
\keyeq{\begin{aligned}
   \mathbf{v}_1 &= (8,\,2,\,-1,\,0)\\[1mm]
   \mathbf{v}_2 &= (-4,\,12,\,0,\,-1)
  \end{aligned}}

% -------------------------------------------------- general ---------
\topic{La solución general}
\hand{Juntando la particular con todo el núcleo, obtenemos \hl{todas} las soluciones:}
\eqfit{\mathbf{x}=\begin{bmatrix}42\\8\\0\\0\end{bmatrix}
   +\lambda_1\begin{bmatrix}8\\2\\-1\\0\end{bmatrix}
   +\lambda_2\begin{bmatrix}-4\\12\\0\\-1\end{bmatrix},\quad \lambda_1,\lambda_2\in\mathbb{R}}
\obs{$2$ parámetros libres $=4$ incógnitas $-\,2$ pivotes.}

% -------------------------------------------------- receta ----------
\topic{La receta, en 3 pasos}
\hand{Siempre igual:}
\eqx{\textbf{1.}\ \text{una solución de } A\mathbf{x}=\mathbf{b}}
\eqx{\textbf{2.}\ \text{todas las de } A\mathbf{x}=\mathbf{0}}
\eqx{\textbf{3.}\ \text{súmalas}}
\obs{Ni la particular ni la general son únicas — hay libertad en cómo las eliges.}

% -------------------------------------------------- estructura ------
\topic[80mm]{¿Qué forma tiene?}
\hand{El conjunto solución es un \hl{subespacio afín}: el núcleo (que pasa por $\mathbf{0}$) \hl{trasladado} hasta $\mathbf{x}_p$.}
\begin{center}
\begin{tikzpicture}[x=1mm, y=1mm, >=stealth]
  % núcleo: recta por el origen (punteada)
  \draw[accent, densely dashed, line width=1pt] (-34,-17) -- (30,15);
  \node[accent, font=\scriptsize, anchor=west] at (30,15) {$A\mathbf{x}=\mathbf{0}$};
  % conjunto solución: recta paralela desplazada
  \draw[eqink, line width=1.3pt] (-34,3) -- (30,35);
  \node[eqink, font=\scriptsize, anchor=west] at (30,35) {$A\mathbf{x}=\mathbf{b}$};
  % origen
  \fill[ink] (0,0) circle (0.9pt);
  \node[ink, font=\scriptsize, anchor=north] at (0,-2) {$\mathbf{0}$};
  % x_p: flecha del origen a la recta solución
  \draw[ink, -{Latex[length=1.8mm]}, line width=0.9pt] (0,0) -- (-8,17);
  \node[ink, font=\scriptsize, anchor=east] at (-8,17) {$\mathbf{x}_p$};
\end{tikzpicture}
\end{center}
\obs{Un \hl{punto} ($\mathbf{x}_p$) más una \hl{dirección} entera (el núcleo).}

% -------------------------------------------------- ¿y si no? -------
\topic{¿Y si no es tan fácil?}
\hand{Aquí fue directo porque la matriz tenía la forma $[\,I \mid *\,]$. Para cualquier otro sistema, la eliminación \hl{gaussiana} lo lleva justo a esa forma — y ahí aplicamos la misma receta.}
\obs{Ese es el tema de la siguiente parte.}
