% -------------------------------------------------- qué es ---------
\topic{¿Qué es el rango?}
\hand{El \uhand{rango} $\operatorname{rk}(A)$ es el número de columnas linealmente \hl{independientes} de $A$ —y resulta ser también el de filas independientes.}
\keyeq{\operatorname{rk}(A) = \operatorname{rk}(A^\top)}
\hand{Se lee contando \hl{pivotes} tras la eliminación gaussiana:}
\begin{center}
\begin{tikzpicture}[x=6.6mm, y=6.6mm]
  \cell{0}{0}{hlA}{$1$}\cell{1}{0}{cellBase}{$2$}\cell{2}{0}{cellBase}{$1$}
  \cell{0}{-1}{cellBase}{$0$}\cell{1}{-1}{hlA}{$1$}\cell{2}{-1}{cellBase}{$3$}
  \cell{0}{-2}{cellNull}{$0$}\cell{1}{-2}{cellNull}{$0$}\cell{2}{-2}{cellNull}{$0$}
  \node[font=\scriptsize\color{ink}, anchor=west] at (2.8,-1){$\operatorname{rk}=2$};
\end{tikzpicture}
\end{center}
\obs{Dos pivotes (aqua) y una fila nula dan rango $2$.}

% -------------------------------------------------- es dimensión ---
\topic{El rango es una dimensión}
\hand{Las columnas de $A$ generan un subespacio —la \hl{imagen} de $A$— y su dimensión es exactamente el rango:}
\keyeq{\dim(\operatorname{im} A) = \operatorname{rk}(A)}
\obs{Lo mismo vale por filas. Por eso el rango mide cuánta información \hl{no redundante} carga la matriz.}

% -------------------------------------------------- qué dice ------
\topic{¿Qué te dice el rango?}
\begin{center}
\begin{tikzpicture}
  \node[draw=cellEdge, line width=0.4pt, fill=cellBase, rounded corners=1.4mm,
        text width=74mm, inner sep=2.6mm, font=\small\color{ink}] {%
    \textbf{Invertible:}\ $A_{n\times n}$ es regular $\iff \operatorname{rk}(A)=n$.\\[1.5mm]
    \textbf{Resoluble:}\ $A\mathbf{x}=\mathbf{b}$ tiene solución $\iff \operatorname{rk}(A)=\operatorname{rk}([A\mid\mathbf{b}])$.\\[1.5mm]
    \textbf{Núcleo:}\ las soluciones de $A\mathbf{x}=\mathbf{0}$ forman un subespacio de dimensión $n-\operatorname{rk}(A)$.};
\end{tikzpicture}
\end{center}

% -------------------------------------------------- rango completo -
\topic{Rango completo}
\hand{El rango nunca pasa de $\min(m,n)$. Cuando lo alcanza, la matriz tiene \hl{rango completo}:}
\keyeq{\operatorname{rk}(A) = \min(m,n)}
\obs{Si se queda corto, es \hl{deficiente}: hay columnas (o filas) que son combinación de las demás —redundancia pura.}
