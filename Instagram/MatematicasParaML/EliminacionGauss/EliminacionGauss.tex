% -------------------------------------------------- elementales -----
\topic{Las transformaciones elementales}
\hand{Tres operaciones sobre las filas \hl{no cambian} el conjunto solución:}
\begin{center}
\begin{tikzpicture}[x=6mm, y=6mm]
  % ===== 1. intercambiar dos filas =====
  \cell{0}{3.5}{hlA}{$1$}\cell{1}{3.5}{hlA}{$2$}
  \cell{0}{2.5}{hlB}{$3$}\cell{1}{2.5}{hlB}{$4$}
  \draw[-{Latex[length=2mm]}, eqink, line width=0.7pt] (1.8,3) -- (2.7,3);
  \cell{3.5}{3.5}{hlB}{$3$}\cell{4.5}{3.5}{hlB}{$4$}
  \cell{3.5}{2.5}{hlA}{$1$}\cell{4.5}{2.5}{hlA}{$2$}
  \node[font=\scriptsize\color{ink}, anchor=west] at (5.4,3){intercambiar filas};
  % ===== 2. multiplicar una fila por 2 =====
  \cell{0}{0.5}{hlC}{$1$}\cell{1}{0.5}{hlC}{$2$}
  \cell{0}{-0.5}{cellBase}{$3$}\cell{1}{-0.5}{cellBase}{$4$}
  \draw[-{Latex[length=2mm]}, eqink, line width=0.7pt] (1.8,0) -- (2.7,0);
  \cell{3.5}{0.5}{hlC}{$2$}\cell{4.5}{0.5}{hlC}{$4$}
  \cell{3.5}{-0.5}{cellBase}{$3$}\cell{4.5}{-0.5}{cellBase}{$4$}
  \node[font=\scriptsize\color{ink}, anchor=west] at (5.4,0){fila$_1 \times\,2$};
  % ===== 3. sumar un múltiplo de otra (crea un 0) =====
  \cell{0}{-2.5}{cellBase}{$1$}\cell{1}{-2.5}{cellBase}{$2$}
  \cell{0}{-3.5}{hlA}{$3$}\cell{1}{-3.5}{hlA}{$4$}
  \draw[-{Latex[length=2mm]}, eqink, line width=0.7pt] (1.8,-3) -- (2.7,-3);
  \cell{3.5}{-2.5}{cellBase}{$1$}\cell{4.5}{-2.5}{cellBase}{$2$}
  \cell{3.5}{-3.5}{hlA}{$0$}\cell{4.5}{-3.5}{hlA}{$-2$}
  \node[font=\scriptsize\color{ink}, anchor=west] at (5.4,-3){fila$_2 - 3\cdot$fila$_1$};

\end{center}

% -------------------------------------------------- aumentada ------
\topic{La matriz aumentada}
\hand{Guardamos solo los números en la \hl{matriz aumentada} $[A\mid\mathbf{b}]$; las incógnitas quedan implícitas. Sobre ella operamos por filas.}
\begin{center}
\begin{tikzpicture}[x=6.4mm, y=6.4mm]
  \foreach \r in {0,-1}{\foreach \c in {0,1,2}{\cell{\c}{\r}{cellBase}{}}}  % A: cols 0-2
  \cell{3.3}{0}{hlC}{}\cell{3.3}{-1}{hlC}{}                                 % b: adyacente
  \node[font=\footnotesize\color{ink}] at (1,1) {$A$};
  \node[font=\footnotesize\color{ink}] at (3.3,1) {$\mathbf{b}$};
  % barra separadora justo entre A y b
  \draw[ink, line width=1pt] (2.65,0.58) -- (2.65,-1.58);
  % corchetes alrededor de todo [A|b]
  \draw[ink, line width=0.7pt] (-0.6,0.62)--(-0.78,0.62)--(-0.78,-1.62)--(-0.6,-1.62);
  \draw[ink, line width=0.7pt] (3.9,0.62)--(4.08,0.62)--(4.08,-1.62)--(3.9,-1.62);
\end{tikzpicture}
\end{center}

% -------------------------------------------------- escalonada -----
\topic{Hasta la forma escalonada}
\hand{Aplicamos las operaciones hasta dejar una \hl{escalera}: la forma escalonada por filas.}
\begin{center}
\begin{tikzpicture}[x=6.4mm, y=6.4mm]
  % REF 4x5, pivotes en (0,0),(2,-1),(3,-2)
  \cell{0}{0}{hlA}{$1$}\cell{1}{0}{cellBase}{$*$}\cell{2}{0}{cellBase}{$*$}\cell{3}{0}{cellBase}{$*$}\cell{4}{0}{cellBase}{$*$}
  \cell{0}{-1}{cellBase}{$0$}\cell{1}{-1}{cellBase}{$0$}\cell{2}{-1}{hlA}{$1$}\cell{3}{-1}{cellBase}{$*$}\cell{4}{-1}{cellBase}{$*$}
  \cell{0}{-2}{cellBase}{$0$}\cell{1}{-2}{cellBase}{$0$}\cell{2}{-2}{cellBase}{$0$}\cell{3}{-2}{hlA}{$1$}\cell{4}{-2}{cellBase}{$*$}
  \cell{0}{-3}{cellBase}{$0$}\cell{1}{-3}{cellBase}{$0$}\cell{2}{-3}{cellBase}{$0$}\cell{3}{-3}{cellBase}{$0$}\cell{4}{-3}{cellBase}{$0$}
  % la escalera
  \draw[eqink, line width=1.2pt]
    (-0.5,0.5)--(-0.5,-0.5)--(1.5,-0.5)--(1.5,-1.5)--(2.5,-1.5)--(2.5,-2.5)--(4.5,-2.5);
\end{tikzpicture}
\end{center}
\obs{Cada \hl{pivote} (primer no-cero de su fila) va estrictamente a la derecha del de arriba; las filas nulas, al fondo.}

% -------------------------------------------------- básicas/libres --
\topic{Básicas y libres}
\hand{Las columnas \hl{con} pivote dan las variables \hl{básicas}; las demás, las \hl{libres} (los parámetros $\lambda$).}
\begin{center}
\begin{tikzpicture}[x=6.4mm, y=6.4mm]
  % misma REF, columnas pivote (0,2,3) en aqua, libres (1,4) en oro
  \foreach \r/\vals in {0/{1,*,*,*,*}, -1/{0,0,1,*,*}, -2/{0,0,0,1,*}}{}
  % fila 0
  \cell{0}{0}{hlA}{$1$}\cell{1}{0}{hlB}{$*$}\cell{2}{0}{hlA}{$*$}\cell{3}{0}{hlA}{$*$}\cell{4}{0}{hlB}{$*$}
  \cell{0}{-1}{hlA}{$0$}\cell{1}{-1}{hlB}{$0$}\cell{2}{-1}{hlA}{$1$}\cell{3}{-1}{hlA}{$*$}\cell{4}{-1}{hlB}{$*$}
  \cell{0}{-2}{hlA}{$0$}\cell{1}{-2}{hlB}{$0$}\cell{2}{-2}{hlA}{$0$}\cell{3}{-2}{hlA}{$1$}\cell{4}{-2}{hlB}{$*$}
  \node[font=\scriptsize\color{ink}] at (0,1) {$x_1$};\node[font=\scriptsize\color{ink}] at (1,1) {$x_2$};
  \node[font=\scriptsize\color{ink}] at (2,1) {$x_3$};\node[font=\scriptsize\color{ink}] at (3,1) {$x_4$};
  \node[font=\scriptsize\color{ink}] at (4,1) {$x_5$};
\end{tikzpicture}
\end{center}
\obs{Aquí $x_1,x_3,x_4$ son \hl{básicas} (aqua) y $x_2,x_5$ son \hl{libres} (oro).}

% -------------------------------------------------- reducida -------
\topic{La forma reducida (RREF)}
\hand{Un paso más: cada pivote $=1$ y es el \hl{único} no-cero de su columna. Esa es la forma escalonada \hl{reducida}.}
\begin{center}
\begin{tikzpicture}[x=6.4mm, y=6.4mm]
  % RREF ejemplo (2.49): [1 3 0 0 3; 0 0 1 0 9; 0 0 0 1 -4]
  \cell{0}{0}{hlA}{$1$}\cell{1}{0}{cellBase}{$3$}\cell{2}{0}{cellBase}{$0$}\cell{3}{0}{cellBase}{$0$}\cell{4}{0}{cellBase}{$3$}
  \cell{0}{-1}{cellBase}{$0$}\cell{1}{-1}{cellBase}{$0$}\cell{2}{-1}{hlA}{$1$}\cell{3}{-1}{cellBase}{$0$}\cell{4}{-1}{cellBase}{$9$}
  \cell{0}{-2}{cellBase}{$0$}\cell{1}{-2}{cellBase}{$0$}\cell{2}{-2}{cellBase}{$0$}\cell{3}{-2}{hlA}{$1$}\cell{4}{-2}{cellBase}{$-4$}
\end{tikzpicture}
\end{center}
\obs{Los pivotes (aqua) valen $1$ y su columna está limpia: la solución se lee casi sin cuentas.}

% -------------------------------------------------- por qué --------
\topic{Por qué lo hacemos}
\hand{Desde la forma escalonada, la solución \hl{particular} y la \hl{general} salen casi directo (justo la receta de la parte anterior).}
\obs{Y la eliminación de Gauss es el motor detrás de resolver sistemas, \hl{invertir} matrices y medir el \hl{rango}.}
