% -------------------------------------------------- qué hace ---------
\topic{¿Qué hace un escalar?}
\hand{Multiplicar por un número $\lambda$ \hl{escala} cada entrada de la matriz, una por una:}
\begin{center}
\begin{tikzpicture}[x=6.4mm, y=6.4mm]
  \node[font=\large\color{accent}] at (-1.1,0) {$3\;\cdot$};
  % C = [1 2; 3 4]
  \cell{0}{0.5}{cellBase}{$1$}\cell{1}{0.5}{cellBase}{$2$}
  \cell{0}{-0.5}{cellBase}{$3$}\cell{1}{-0.5}{cellBase}{$4$}
  \node[font=\large\color{eqink}] at (2.5,0) {$=$};
  % 3C = [3 6; 9 12]  (verde: todas escaladas)
  \cell{4}{0.5}{hlC}{$3$}\cell{5}{0.5}{hlC}{$6$}
  \cell{4}{-0.5}{hlC}{$9$}\cell{5}{-0.5}{hlC}{$12$}
\end{tikzpicture}
\end{center}
\keyeq{\lambda A = K,\qquad K_{ij} = \lambda\,a_{ij}}

% -------------------------------------------------- plano -----------
\topic{Su efecto en el plano}
\hand{Como transformación, un escalar \hl{agranda o encoge} todo por igual: el \hl{círculo unitario} se vuelve un círculo de radio $\lambda$.}
\begin{center}
\begin{tikzpicture}[x=7mm, y=7mm, >=stealth]
  % ejes tenues
  \draw[mauve, line width=0.4pt] (-2.5,0) -- (2.5,0);
  \draw[mauve, line width=0.4pt] (0,-2.5) -- (0,2.5);
  % círculo escalado (radio 2, verde) y círculo unitario (punteado)
  \draw[accent, line width=1.3pt] (0,0) circle (2);
  \draw[ink, line width=0.8pt, densely dashed] (0,0) circle (1);
  % un vector v en el círculo unitario -> 2v en el círculo grande
  \draw[ink,    line width=1.1pt, -{Latex[length=2mm]}]   (0,0) -- (0.707,0.707);
  \draw[accent, line width=1.3pt, -{Latex[length=2.2mm]}] (0,0) -- (1.414,1.414);
  \draw[amber,  line width=1.1pt, -{Latex[length=2mm]}]   (0,0) -- (-0.707,-0.707);
  \node[accent, font=\scriptsize, anchor=south west] at (1.35,1.5)  {$2\vec v$};
  \node[ink,    font=\scriptsize, anchor=north west] at (0.72,0.6)  {$\vec v$};
  \node[amber,  font=\scriptsize, anchor=north east] at (-0.72,-0.72) {$-\vec v$};
\end{tikzpicture}
\end{center}
\obs{Con $\lambda>1$ crece, con $0<\lambda<1$ se encoge, y con $\lambda<0$ se refleja. Una matriz cualquiera lo deformaría en una \hl{elipse}; el escalar solo lo escala parejo.}

% -------------------------------------------------- se desliza ------
\topic{El escalar se desliza}
\hand{Dentro de un producto, el escalar puede ponerse en \hl{cualquier lugar}: el resultado no cambia.}
\begin{center}
\begin{tikzpicture}[x=1mm, y=1mm]
  \node[font=\Large] at (0,0)
    {$\textcolor{accent}{\lambda}BC = B\textcolor{accent}{\lambda}C = BC\textcolor{accent}{\lambda}$};
  \draw[accent, densely dashed, -{Latex[length=1.6mm]}] (-30,9) to[bend left=20] (30,9);
  \node[font=\scriptsize\itshape\color{ink}] at (0,16) {el escalar se desliza};
\end{tikzpicture}
\end{center}
\obs{También se combinan entre sí: $(\lambda\psi)C = \lambda(\psi C)$.}

% -------------------------------------------------- distributividad -
\topic{Se reparte sobre las sumas}
\hand{El escalar se \hl{distribuye}: sobre una suma de escalares, y sobre una suma de matrices.}
\begin{center}
\begin{tikzpicture}[x=1mm, y=1mm]
  \node[font=\Large] at (0,0)
    {$(\textcolor{accent}{\lambda}{+}\textcolor{amber}{\psi})\,C
       = \textcolor{accent}{\lambda}C + \textcolor{amber}{\psi}C$};
  \draw[accent, -{Latex[length=1.4mm]}, line width=0.5pt] (-26,-8) to[bend right=25] (7,-8);
  \draw[amber,  -{Latex[length=1.4mm]}, line width=0.5pt] (-17,9) to[bend left=25] (30,9);
\end{tikzpicture}
\end{center}
\eqx{\lambda(B+C) = \lambda B + \lambda C}

% -------------------------------------------------- transpuesta -----
\topic{Pasa por la transpuesta}
\hand{Un escalar \hl{atraviesa} la transpuesta sin inmutarse (al fin y al cabo, $\lambda = \lambda^{\top}$):}
\keyeq{(\lambda C)^{\top} = \lambda\,C^{\top}}

% -------------------------------------------------- ejemplo ---------
\topic{Un ejemplo}
\eg{con $C = \begin{bmatrix}1&2\\3&4\end{bmatrix}$, escalar por $2$ duplica cada entrada:}
\eqx{2\,C = \begin{bmatrix}2&4\\6&8\end{bmatrix}}
\hand{y repartir da lo mismo: $(2+3)\,C = 2C + 3C = 5C$.}

% -------------------------------------------------- por qué importa -
\topic{¿Por qué importa en ML?}
\hand{Escalar aparece en todo ML: el \hl{tamaño de paso} $\alpha\nabla f$, normalizar datos, o la \hl{regularización} $\lambda\lVert w\rVert^2$.}
