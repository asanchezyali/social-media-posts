% -------------------------------------------------- qué es -------
\topic{¿Qué es?}
\hand{Dado un subespacio $U\subseteq V$, su complemento \uhand{ortogonal} $U^\perp$ reúne \emph{todos} los vectores perpendiculares a $U$:}
\keyeq{U^\perp = \{\,\mathbf{v}\in V : \langle\mathbf{v},\mathbf{u}\rangle=0 \ \forall\,\mathbf{u}\in U\,\}}
\obs{También es un subespacio, y se cortan solo en el origen: $U\cap U^\perp=\{\mathbf{0}\}$.}

% -------------------------------------------------- dimensiones -
\topic{Se reparten el espacio}
\hand{Entre $U$ y $U^\perp$ cubren todo $V$ sin solaparse —una suma \hl{directa}:}
\keyeq{V = U \oplus U^\perp}
\obs{Las dimensiones encajan: $\dim U + \dim U^\perp = \dim V$.}
\obs{Así, cada vector se descompone de forma única: algo en $U$ y algo en $U^\perp$.}

% -------------------------------------------------- ilustración -
\topic{El vector normal}
\hand{En $\mathbb{R}^3$, si $U$ es un plano, su complemento es la \hl{recta normal}: la dirección perpendicular al plano.}
\begin{center}
\begin{tikzpicture}[x=10mm,y=10mm]
  \fill[accent!16] (-1.5,-0.55)--(1.5,-1.0)--(1.5,0.35)--(-1.5,0.8)--cycle;
  \draw[accent!75!ink, line width=0.7pt] (-1.5,-0.55)--(1.5,-1.0)--(1.5,0.35)--(-1.5,0.8)--cycle;
  \node[font=\scriptsize\color{accent}] at (1.15,-0.55){$U$};
  \draw[->, eqink, line width=1.2pt] (0,-0.12)--(0,1.9) node[right,font=\scriptsize\color{eqink}]{$U^\perp$};
  \draw[coral, line width=0.6pt] (0.26,-0.08)--(0.26,0.18)--(0,0.18);
  \fill[ink] (0,-0.12) circle (0.8pt);
\end{tikzpicture}
\end{center}
\obs{Ese vector normal define el plano: es la clave de los \hl{hiperplanos} y de las fronteras de decisión en ML.}
