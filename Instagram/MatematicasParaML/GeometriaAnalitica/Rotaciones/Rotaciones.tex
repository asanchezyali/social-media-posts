% -------------------------------------------------- qué es -------
\topic{¿Qué es una rotación?}
\hand{Es una aplicación lineal \hl{ortogonal} que gira el espacio: preserva longitudes, ángulos y área. Nada se estira ni se refleja.}
\obs{Su matriz $R$ cumple $R^\top R=I$ y $\det R=1$ (una reflexión tendría $\det=-1$).}

% -------------------------------------------------- plano -------
\topic{En el plano}
\hand{Girar un ángulo $\theta$ en $\mathbb{R}^2$ se hace con una sola matriz:}
\keyeq{R(\theta)=\begin{bmatrix}\cos\theta & -\sin\theta\\[1mm] \sin\theta & \cos\theta\end{bmatrix}}
\begin{center}
\begin{tikzpicture}[x=11mm,y=11mm,>={Latex[length=2mm]}]

  % base original
  \draw[->, mauve, line width=1pt] (0,0)--(1.6,0) node[right,font=\tiny\color{mauve}]{$\mathbf{e}_1$};
  % base rotada 40 grados
  \draw[->, accent, line width=1.1pt] (0,0)--(1.226,1.028) node[right,font=\scriptsize\color{accent}]{$R\mathbf{e}_1$};
  \draw[->, eqink, line width=1.1pt]  (0,0)--(-1.028,1.226) node[above,font=\scriptsize\color{eqink}]{$R\mathbf{e}_2$};
  \draw[coral, line width=0.8pt] (0.7,0) arc (0:40:0.7);
  \node[font=\scriptsize\color{coral}] at (0.92,0.28){$\theta$};
  \fill[ink] (0,0) circle (0.8pt);
\end{tikzpicture}
\end{center}

% -------------------------------------------------- propiedades -
\topic{Sus propiedades}
\obs{La inversa es \hl{gratis}: $R(\theta)^{-1}=R(\theta)^\top=R(-\theta)$, girar hacia el otro lado. Y componer rotaciones es sumar ángulos.}
\obs{En $\mathbb{R}^3$ hay una rotación por cada eje; combinadas giran cualquier objeto en el espacio.}

% -------------------------------------------------- ML ---------
\topic{En machine learning}
\obs{Las rotaciones cambian de \hl{coordenadas} sin perder información y generan datos (\emph{data augmentation}): la misma imagen, vista desde otro ángulo.}
