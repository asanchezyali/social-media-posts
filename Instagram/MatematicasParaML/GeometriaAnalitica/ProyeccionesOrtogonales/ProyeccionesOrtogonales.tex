% -------------------------------------------------- idea --------
\topic{La idea}
\hand{Proyectar $\mathbf{x}$ sobre un subespacio $U$ es hallar el punto de $U$ \hl{más cercano} a $\mathbf{x}$. La clave: el error $\mathbf{x}-\pi_U(\mathbf{x})$ queda \hl{perpendicular} a $U$.}
\obs{Es comprimir con la menor pérdida posible —el corazón de PCA, mínimos cuadrados y la regresión.}

% -------------------------------------------------- sobre recta -
\topic{Sobre una recta}
\hand{Si $U$ es la recta generada por $\mathbf{b}$, la sombra de $\mathbf{x}$ es:}
\keyeq{\pi_U(\mathbf{x}) = \dfrac{\langle\mathbf{x},\mathbf{b}\rangle}{\langle\mathbf{b},\mathbf{b}\rangle}\,\mathbf{b}}
\begin{center}
\begin{tikzpicture}[x=11mm,y=11mm,>={Latex[length=2mm]}]
  \draw[mauve, line width=0.9pt] (-0.5,-0.35)--(2.7,1.89) node[right,font=\scriptsize\color{mauve}]{$U=\operatorname{span}[\mathbf{b}]$};
  \coordinate (X) at (1.0,1.9);
  \coordinate (P) at (1.732,1.212);   % pie de la perpendicular sobre la recta
  \draw[->, ink, line width=1.1pt] (0,0)--(X) node[left,font=\scriptsize\color{ink}]{$\mathbf{x}$};
  \draw[->, accent, line width=1.1pt] (0,0)--(P) node[below right,font=\scriptsize\color{accent}]{$\pi_U(\mathbf{x})$};
  \draw[coral, densely dashed, line width=0.7pt] (X)--(P);
  \draw[coral, line width=0.6pt] ($(P)+(-0.22,0.06)$)--($(P)+(-0.15,0.28)$)--($(P)+(0.07,0.21)$);
  \fill[ink] (0,0) circle (0.8pt);
\end{tikzpicture}
\end{center}

% -------------------------------------------------- matriz -----
\topic{La matriz de proyección}
\hand{La proyección es \hl{lineal}: se aplica con una matriz. Sobre una recta y sobre un subespacio general $U=\operatorname{span}[B]$:}
\keyeq{P = \dfrac{\mathbf{b}\mathbf{b}^\top}{\mathbf{b}^\top\mathbf{b}}, \qquad P = B(B^\top B)^{-1}B^\top}
\obs{$P$ es simétrica y cumple $P^2=P$ : proyectar dos veces da lo mismo.}

% -------------------------------------------------- ML ---------
\topic{¿Por qué importa en ML?}
\obs{Mínimos cuadrados proyecta $\mathbf{b}$ sobre el espacio columna de $A$; PCA proyecta los datos sobre las direcciones de mayor \hl{varianza}. Comprimir bien \emph{es} proyectar bien.}
