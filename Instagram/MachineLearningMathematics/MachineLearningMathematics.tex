\topic{Descenso de gradiente}
\hand{El paso de primer orden avanza en contra del gradiente, escalado por un tamaño de paso $\alpha$:}
\eq{x_{k+1} = x_k - \alpha \nabla f(x_k)}
\obs{solo usa la pendiente, no cómo está cambiando.}

\topic{Un paso más fino}
\hand{Un dato extra hace cada paso más inteligente:}
\keyeq{x_{k+1} = x_k - \hlm{H^{-1}} \nabla f(x_k)}
\hand{...y solo necesita un poco del cálculo que ya conocemos.}

\topic{Curvatura}
\defn{La \uhand{expansión de Taylor} ajusta una parábola a $f$ cerca de un punto $a$:}
\eq{\begin{aligned}
    f(x) &= f(a) + \frac{f'(a)}{1!}(x-a)\\[1mm]
         &\;+ \frac{f''(a)}{2!}(x-a)^2 + \dots
  \end{aligned}}
\obs{el término cuadrático lleva la \hl{curvatura}.}

\topic{Método de Newton (1D)}
\hand{Sea $a = x_k$ y $x-a = t$, y minimizamos la parábola:}
\eqx{f(x_k+t) \approx f(x_k) + f'(x_k)t + \tfrac{1}{2}f''(x_k)t^2}
\eqx{\Rightarrow\quad t = -\frac{f'(x_k)}{f''(x_k)}}

\topic{Ejemplo}
\eg{para $f(x) = 0.05x^4 - 0.2x^3 - 0.5x^2 + x + 2$ en $x_k = 3.5$:}
\eqx{f'(3.5) = -1.275,\; f''(3.5) = 2.15}
\eqx{t = -\frac{(-1.275)}{2.15} \approx 0.593 \Rightarrow x_{k+1} \approx 4.093}

\topic{Muchos parámetros}
\hand{Los modelos reales tienen muchos parámetros, así que $f'$ y $f''$ pasan a sus análogos multivariados, hechos de derivadas parciales.}
\defn{El gradiente \uhand{$\nabla f$} es el vector de todas las primeras derivadas parciales:}
\eq{\nabla f = \begin{bmatrix} \dfrac{\partial f}{\partial x_1} \\[4mm]
    \dfrac{\partial f}{\partial x_2} \\[3.5mm] \vdots \end{bmatrix}}

\topic{La Hessiana}
\defn{La Hessiana \uhand{$H$} es la matriz de todas las segundas derivadas parciales:}
\eqfit{H_f = \begin{bmatrix}
    \dfrac{\partial^2 f}{\partial x_1^2} & \dfrac{\partial^2 f}{\partial x_1 \partial x_2} & \cdots & \dfrac{\partial^2 f}{\partial x_1 \partial x_n} \\[5mm]
    \dfrac{\partial^2 f}{\partial x_2 \partial x_1} & \dfrac{\partial^2 f}{\partial x_2^2} & \cdots & \dfrac{\partial^2 f}{\partial x_2 \partial x_n} \\[4.5mm]
    \vdots & \vdots & \ddots & \vdots \\[4.5mm]
    \dfrac{\partial^2 f}{\partial x_n \partial x_1} & \dfrac{\partial^2 f}{\partial x_n \partial x_2} & \cdots & \dfrac{\partial^2 f}{\partial x_n^2}
  \end{bmatrix}}

\topic{Newton en $n$ dimensiones}
\hand{Reemplazamos $f' \to \nabla f$ y $f'' \to H$:}
\eq{x_{k+1} = x_k - \frac{f'(x_k)}{f''(x_k)}}
\keyeq{x_{k+1} = x_k - \hlm{H^{-1}} \nabla f(x_k)}

\topic{En la práctica}
\obs{calcular la Hessiana completa es costoso en modelos con muchas dimensiones.}
\hand{Por eso normalmente se \hl{aproxima} $H^{-1}$ — la idea detrás de los métodos quasi-Newton como BFGS.}

\vspace{2mm}
\hand{En el fondo, dos ideas antiguas — la expansión de \hl{Taylor} y el paso de \hl{Newton} — todavía dan forma a los algoritmos que entrenan los modelos de hoy.}
