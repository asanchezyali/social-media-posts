% ===================== 01 · SELECT ==================================
\begin{sld}[01]
  \kicker{\kblue{01} · read data}
  \pill{SELECT}
  \headline{Pick \hb{which columns}\\ you want}
  \lead{Ask a table for the columns you need. \chip{*} means \tb{all of them}.}
  \begin{codecard}{select.sql}
-- all columns
SELECT * FROM users;
-- only the ones you need
SELECT name, email FROM users;
  \end{codecard}
\end{sld}

% ===================== 02 · WHERE ===================================
\begin{sld}[02]
  \kicker{\kblue{02} · filter rows}
  \pill{WHERE}
  \headline{Keep only the rows that \hb{match}}
  \lead{Like a search box inside the table: \tb{filter} and keep just what you care about.}
  \begin{codecard}{where.sql}
SELECT * FROM users
WHERE age >= 18
  AND country = 'MX';
  \end{codecard}
\end{sld}

% ===================== 03 · ORDER BY ================================
\begin{sld}[03]
  \kicker{\kblue{03} · sort}
  \pill{ORDER BY}
  \headline{Sort the result \hb{your way}}
  \lead{Ascending by default; add \chip{DESC} for \tb{highest first}.}
  \begin{codecard}{order.sql}
SELECT name, age FROM users
ORDER BY age DESC;
  \end{codecard}
\end{sld}

% ===================== 04 · LIMIT ===================================
\begin{sld}[04]
  \kicker{\kblue{04} · cap rows}
  \pill{LIMIT}
  \headline{Ask for\\ \hb{just a few} rows}
  \lead{Perfect for \tb{top-N} lists and fast table previews.}
  \begin{codecard}{limit.sql}
SELECT * FROM users
ORDER BY created_at DESC
LIMIT 10;
  \end{codecard}
\end{sld}

% ===================== 05 · JOIN ====================================
\begin{sld}[05]
  \kicker{\kblue{05} · combine tables}
  \pill{JOIN}
  \headline{Glue two tables\\ by a \hb{shared column}}
  \lead{Data lives split (users, orders). JOIN connects them by a key, like \chip{user\_id}.}
  \begin{codecard}{join.sql}
SELECT u.name, o.total
FROM users u
JOIN orders o
  ON o.user_id = u.id;
  \end{codecard}
\end{sld}

% ===================== 06 · GROUP BY ================================
\begin{sld}[06]
  \kicker{\kblue{06} · aggregate}
  \pill{GROUP BY}
  \headline{Collapse rows into \hb{one per group}}
  \lead{Pair it with \chip{COUNT}, \chip{SUM} or \chip{AVG} to \tb{summarize} your data.}
  \begin{codecard}{group.sql}
SELECT country, COUNT(*)
FROM users
GROUP BY country;
  \end{codecard}
\end{sld}

% ===================== 07 · LIKE ====================================
\begin{sld}[07]
  \kicker{\kblue{07} · search text}
  \pill{LIKE}
  \headline{Match text\\ by a \hb{pattern}}
  \lead{\chip{\%} is a wildcard: \chip{'A\%'} means \tb{starts with A}.}
  \begin{codecard}{like.sql}
SELECT * FROM users
WHERE name LIKE 'A%';
  \end{codecard}
\end{sld}

% ===================== 08 · INSERT ==================================
\begin{sld}[08]
  \kicker{\kblue{08} · add data}
  \pill{INSERT}
  \headline{Add a \hb{new row} to a table}
  \lead{List the columns, then the \tb{values} that fill them.}
  \begin{codecard}{insert.sql}
INSERT INTO users (name, age)
VALUES ('Ada', 36);
  \end{codecard}
\end{sld}

% ===================== 09 · UPDATE ==================================
\begin{sld}[09]
  \kicker{\kblue{09} · change data}
  \pill{UPDATE}
  \headline{Edit rows that\\ \hb{already exist}}
  \lead{No \chip{WHERE} means you change \tb{every row}. Never forget it.}
  \begin{codecard}{update.sql}
UPDATE users
SET age = 37
WHERE id = 1;
  \end{codecard}
\end{sld}

% ===================== 10 · DELETE ==================================
\begin{sld}[10]
  \kicker{\kblue{10} · remove data}
  \pill{DELETE}
  \headline{Remove rows you \hb{no longer need}}
  \lead{Same rule: no \chip{WHERE}, and the \tb{whole table} is gone.}
  \begin{codecard}{delete.sql}
DELETE FROM users
WHERE id = 1;
  \end{codecard}
\end{sld}
