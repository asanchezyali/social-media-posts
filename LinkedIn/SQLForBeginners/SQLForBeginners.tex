% ===================== CONTEXT ======================================
\begin{sld}
  \kicker{\kgreen{before the syntax}}
  \headlinesm{SQL is the skill that\\ \hb{outlives frameworks}}
  \body{Frameworks come and go, but relational databases have run the world's data for 50 years. Learn to think in \tb{sets} and the same model works everywhere.}
  \pt{It's \tb{declarative} — you say what you want, the engine finds how.}
  \pt{Indexes and the optimizer do the heavy lifting, \tb{if you let them}.}
  \begin{callout}{WHY IT MATTERS}
  Backend, analytics, ML — every data role runs on SQL. You rarely get to avoid it.
  \end{callout}
\end{sld}

% ===================== 01 · SELECT ==================================
\begin{sld}
  \kicker{\kblue{01} · read data}
  \headlinesm{Pick \hb{which columns} you want}
  \begin{codecard}{select.sql}
SELECT * FROM users;
SELECT id, name FROM users;
  \end{codecard}
  \begin{callout}[dotamber]{GOTCHA}
  \chip{SELECT *} ships every column and breaks when the schema changes. Name the columns you actually use in production.
  \end{callout}
\end{sld}

% ===================== 02 · WHERE ===================================
\begin{sld}
  \kicker{\kblue{02} · filter rows}
  \headlinesm{Keep only the rows that \hb{match}}
  \begin{codecard}{where.sql}
SELECT * FROM users
WHERE country = 'MX'
  AND last_login IS NOT NULL;
  \end{codecard}
  \begin{callout}[dotamber]{GOTCHA · NULL}
  \chip{= NULL} never matches — NULL means "unknown". Use \chip{IS NULL}. And a function on the column blocks the index.
  \end{callout}
\end{sld}

% ===================== 03 · ORDER BY ================================
\begin{sld}
  \kicker{\kblue{03} · sort}
  \headlinesm{Sort the result \hb{your way}}
  \begin{codecard}{order.sql}
SELECT * FROM users
ORDER BY created_at DESC, id;
  \end{codecard}
  \begin{callout}[dotamber]{GOTCHA}
  Ties sort in an undefined order — add a tiebreaker like \chip{id}. Sorting big results is costly unless the column is indexed.
  \end{callout}
\end{sld}

% ===================== 04 · LIMIT ===================================
\begin{sld}
  \kicker{\kblue{04} · paginate}
  \headlinesm{Ask for \hb{just a few} rows}
  \begin{codecard}{limit.sql}
SELECT * FROM users
ORDER BY id
LIMIT 20;
  \end{codecard}
  \begin{callout}[dotgreen]{PRO TIP}
  \chip{OFFSET n} re-scans n rows on every page. For deep pagination, filter by the last id (\chip{WHERE id > ...}) instead.
  \end{callout}
\end{sld}

% ===================== 05 · JOIN ====================================
\begin{sld}
  \kicker{\kblue{05} · combine tables}
  \headlinesm{Glue two tables\\ by a \hb{shared key}}
  \begin{codecard}{join.sql}
SELECT u.name, o.total
FROM users u
LEFT JOIN orders o ON o.user_id = u.id;
  \end{codecard}
  \begin{callout}[dotamber]{GOTCHA}
  INNER drops unmatched rows; LEFT keeps them as NULL. Many matches on the right \tb{fan out} into duplicates — aggregate for one row per user.
  \end{callout}
\end{sld}

% ===================== 06 · GROUP BY ================================
\begin{sld}
  \kicker{\kblue{06} · aggregate}
  \headlinesm{Collapse rows into \hb{one per group}}
  \begin{codecard}{group.sql}
SELECT country, COUNT(*)
FROM users
GROUP BY country
HAVING COUNT(*) > 100;
  \end{codecard}
  \begin{callout}[dotamber]{GOTCHA}
  Every selected column must be aggregated or in GROUP BY. \chip{WHERE} filters rows before grouping; \chip{HAVING} filters groups after.
  \end{callout}
\end{sld}

% ===================== 07 · LIKE ====================================
\begin{sld}
  \kicker{\kblue{07} · search text}
  \headlinesm{Match text by a \hb{pattern}}
  \begin{codecard}{like.sql}
SELECT * FROM users
WHERE name LIKE 'A%';
  \end{codecard}
  \begin{callout}[dotamber]{GOTCHA}
  A leading wildcard (\chip{'\%term'}) can't use an index and scans the whole table. For real search, reach for full-text or trigram indexes.
  \end{callout}
\end{sld}

% ===================== 08 · INSERT =================================
\begin{sld}
  \kicker{\kblue{08} · add data}
  \headlinesm{Add \hb{new rows} to a table}
  \begin{codecard}{insert.sql}
INSERT INTO users (name, email) VALUES
  ('Ada',  'ada@mail.com'),
  ('Alan', 'alan@mail.com');
  \end{codecard}
  \begin{callout}[dotgreen]{PRO TIP}
  Insert many rows in one statement, not a loop — one round-trip instead of thousands.
  \end{callout}
\end{sld}

% ===================== 09 · UPDATE / DELETE ========================
\begin{sld}
  \kicker{\kblue{09} · change \& remove}
  \headlinesm{Edit or delete \hb{existing rows}}
  \begin{codecard}{write.sql}
UPDATE users SET active = false
WHERE last_login < '2024-01-01';

DELETE FROM users WHERE id = 42;
  \end{codecard}
  \begin{callout}[dotamber]{GOTCHA}
  No \chip{WHERE} means \tb{every row}. Run your filter as a \chip{SELECT} first to see exactly what you'll touch.
  \end{callout}
\end{sld}

% ===================== COMMON MISTAKES =============================
\begin{sld}
  \kicker{\kgreen{what trips people up}}
  \headlinesm{The mistakes that \hb{cost you hours}}
  \pt{\chip{IS NULL} matches nulls — \chip{= NULL} never does.}
  \pt{No \chip{WHERE} on an UPDATE or DELETE hits \tb{every row}.}
  \pt{\chip{SELECT *} in production ships columns you don't need.}
  \pt{Functions on indexed columns \tb{quietly kill} performance.}
  \begin{callout}[dotgreen]{PRO TIP}
  Read the plan with \chip{EXPLAIN} before blaming the database. Most "slow SQL" is a missing index.
  \end{callout}
\end{sld}
