% =====================================================================
%  Page chrome  (drawn behind content on EVERY page):
%   glow · line-number gutter · giant watermark number · progress dots
% =====================================================================
\newcommand{\DeckChrome}{%
  \begin{tikzpicture}[remember picture, overlay]
    \fill[bg] (current page.south west) rectangle (current page.north east);
    \begin{scope}
      \clip (current page.south west) rectangle (current page.north east);
      % soft glow in the top-right
      \shade[inner color=bgglow, outer color=bg]
        ([shift={(6mm,-2mm)}]current page.north east) circle (78mm);
    \end{scope}
    % line-number gutter, faint, down the left edge (tight editor spacing)
    \foreach \n [count=\i from 0, evaluate=\i as \yy using -10-\i*5.3] in {2,...,27}{%
      \node[anchor=east, text=gutter, inner sep=0pt]
        at ([xshift=9mm, yshift=\yy mm]current page.north west)
        {\mono\fontsize{7}{7}\selectfont \n};
    }
    \draw[gutterline, line width=0.4pt]
      ([xshift=13mm]current page.north west) --
      ([xshift=13mm]current page.south west);
    % progress dots, bottom centre
    \begin{scope}[shift={([yshift=10mm]current page.south)}]
      \pgfmathsetmacro\total{\decktotal}
      \foreach \d in {1,...,\total}{%
        \pgfmathsetmacro\xx{(\d-(\total+1)/2)*3}
        \pgfmathsetmacro\xa{\xx-1}\pgfmathsetmacro\xb{\xx+1}
        \ifnum\d=\value{deckpage}
          \fill[accentblue, rounded corners=0.55mm]
            (\xa mm,-0.55mm) rectangle (\xb mm,0.55mm);
        \else
          \fill[dotoff] (\xx mm,0) circle (0.55mm);
        \fi
      }
    \end{scope}
  \end{tikzpicture}%
}
\AddToShipoutPictureBG{\DeckChrome}

% =====================================================================
%  Slide environment
%    \begin{sld}[<watermark number>] ... \end{sld}
% =====================================================================
% NOTE: \currentwm is set GLOBALLY (\gdef): a page ships at the NEXT slide's
% \clearpage, after this environment's group has closed, so a local value would
% already be gone by shipout.
% Content is anchored to the top, so the headline lands in the same place on
% every swipe (a consistent read anchor across the carousel).
\newenvironment{sld}[1][]{%
  \clearpage\stepcounter{deckpage}%
  \gdef\currentwm{#1}%
  \RaggedRight\setlength{\parskip}{0pt}%
}{\par}
% alias kept for the cover (same behaviour; named for intent)
\newenvironment{sldt}[1][]{%
  \clearpage\stepcounter{deckpage}%
  \gdef\currentwm{#1}%
  \RaggedRight\setlength{\parskip}{0pt}%
}{\par}

  language=SQL,
  basicstyle=\mono\fontsize{9.5}{13.5}\selectfont\color{codeident},
  keywordstyle=\monob\color{codekw},
  commentstyle=\mono\color{codecomment},
  stringstyle=\mono\color{codestring},
  showstringspaces=false, columns=fullflexible, keepspaces=true,
  breaklines=true, tabsize=2,
  morekeywords={JOIN,ON,LIMIT,ORDER,BY,GROUP,HAVING,INNER,LEFT,RIGHT,OUTER,
                AS,DISTINCT,INSERT,INTO,VALUES,UPDATE,DELETE,CREATE,TABLE},
  literate=%
    {0}{{\color{codenum}0}}1 {1}{{\color{codenum}1}}1 {2}{{\color{codenum}2}}1
    {3}{{\color{codenum}3}}1 {4}{{\color{codenum}4}}1 {5}{{\color{codenum}5}}1
    {6}{{\color{codenum}6}}1 {7}{{\color{codenum}7}}1 {8}{{\color{codenum}8}}1
    {9}{{\color{codenum}9}}1 {->}{{\color{codecomment}$\to$}}2
    {*}{{\raisebox{0.45ex}{*}}}1,
}
\newtcblisting{codecard}[1]{%
  listing only, listing style=sqlstyle,

  left=5mm, right=4mm, top=8.4mm, bottom=3.4mm,
  overlay={%
    \fill[dotblue]  ([xshift=5mm,yshift=-4.3mm]frame.north west) circle (1mm);
    \fill[dotamber] ([xshift=8mm,yshift=-4.3mm]frame.north west) circle (1mm);
    \fill[dotgreen] ([xshift=11mm,yshift=-4.3mm]frame.north west) circle (1mm);

      at ([xshift=15mm,yshift=-4.3mm]frame.north west)

      ([yshift=-7.6mm]frame.north west) -- ([yshift=-7.6mm]frame.north east);
  },
